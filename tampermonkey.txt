        updateStatusElement(message);

        if (statusEl) {

        const statusEl = document.getElementById('vps-renewal-progress');

    function waitForDOMReady() {
        return new Promise(resolve => {
            if (document.readyState === 'loading') {
                document.addEventListener('DOMContentLoaded', resolve, { once: true });
            } else {
                resolve();
            }
        });
    }

    function getAccounts() {
        const value = GM_getValue(ACCOUNTS_KEY, []);
        return Array.isArray(value) ? value : [];
    }

    function saveAccounts(accounts) {
        GM_setValue(ACCOUNTS_KEY, accounts);
    }

    function getCurrentAccountIndex() {
        const accounts = getAccounts();
        const index = Number(GM_getValue(CURRENT_INDEX_KEY, 0));
        if (!accounts.length) return 0;
        if (!Number.isInteger(index) || index < 0 || index >= accounts.length) return 0;
        return index;
    }

    function setCurrentAccountIndex(index) {
        GM_setValue(CURRENT_INDEX_KEY, Number(index) || 0);
    }

    function getCurrentAccount() {
        const accounts = getAccounts();
        const index = getCurrentAccountIndex();
        return accounts[index] || null;
    }

    function isAutoLoginEnabled() {
        return GM_getValue(AUTO_LOGIN_KEY, false) === true;
    }

    function setAutoLoginEnabled(enabled) {
        GM_setValue(AUTO_LOGIN_KEY, !!enabled);
    }

    function getLoginInputs() {
        const memberInput = document.querySelector('#memberid, input[name="memberid"]');
        const passInput = document.querySelector('#user_password, input[name="user_password"]');
        return { memberInput, passInput };
    }

    function getInputValue(el) {
        return el ? String(el.value || '').trim() : '';
    }

    function setNativeValue(el, value) {
        if (!el) return;
        el.focus();

        const prototype = Object.getPrototypeOf(el);
        const descriptor = Object.getOwnPropertyDescriptor(prototype, 'value');

        if (descriptor && descriptor.set) {
            descriptor.set.call(el, value);
        } else {
            el.value = value;
        }

        el.dispatchEvent(new Event('input', { bubbles: true }));
        el.dispatchEvent(new Event('change', { bubbles: true }));
        el.dispatchEvent(new KeyboardEvent('keyup', { bubbles: true }));
    }

    function fillLoginFormWithAccount(acc) {
        if (!acc) return false;

        const { memberInput, passInput } = getLoginInputs();

        if (memberInput) {
            setNativeValue(memberInput, acc.memberid || '');
        }

        if (passInput) {
            setNativeValue(passInput, acc.user_password || '');
        }

        return !!(memberInput && passInput);
    }

    function fillLoginFormWithCurrentAccount() {
        return fillLoginFormWithAccount(getCurrentAccount());
    }

    function addOrUpdateAccount(memberid, user_password) {
        const accounts = getAccounts();

        const cleanMemberid = String(memberid || '').trim();
        const cleanPassword = String(user_password || '');

        if (!cleanMemberid || !cleanPassword) {
            throw new Error('账号或密码为空');
        }

        const existingIndex = accounts.findIndex(a => a.memberid === cleanMemberid);

        const item = {
            memberid: cleanMemberid,
            user_password: cleanPassword,
            updated_at: new Date().toISOString(),
        };

        if (existingIndex >= 0) {
            accounts[existingIndex] = {
                ...accounts[existingIndex],
                ...item,
            };
            setCurrentAccountIndex(existingIndex);
        } else {
            accounts.push(item);
            setCurrentAccountIndex(accounts.length - 1);
        }

        saveAccounts(accounts);
        return item;
    }

    function saveCurrentInputAsAccount() {
        const { memberInput, passInput } = getLoginInputs();
        const memberid = getInputValue(memberInput);
        const user_password = passInput ? String(passInput.value || '') : '';

        if (!memberid || !user_password) {
            alert('请先在登录框里输入账号和密码，再保存。');
            return false;
        }

        addOrUpdateAccount(memberid, user_password);
        renderAccountSwitcher();
        showToast('已保存当前输入为账号。');
        return true;
    }

    function deleteCurrentAccount() {
        const accounts = getAccounts();

        if (!accounts.length) {
            alert('没有可删除的账号。');
            return;
        }

        const index = getCurrentAccountIndex();
        const acc = accounts[index];

        if (!confirm(`确定删除当前账号？\n${acc?.memberid || ''}`)) return;

        accounts.splice(index, 1);
        saveAccounts(accounts);
        setCurrentAccountIndex(Math.max(0, index - 1));

        renderAccountSwitcher();
        fillLoginFormWithCurrentAccount();
        showToast('已删除当前账号。');
    }

    function clearAllAccounts() {
        if (!confirm('确定清空所有已保存账号？')) return;

        GM_deleteValue(ACCOUNTS_KEY);
        GM_deleteValue(CURRENT_INDEX_KEY);
        GM_deleteValue(AUTO_LOGIN_KEY);
        GM_deleteValue(SKIP_ONCE_KEY);

        renderAccountSwitcher();

        const { memberInput, passInput } = getLoginInputs();
        if (memberInput) setNativeValue(memberInput, '');
        if (passInput) setNativeValue(passInput, '');

        showToast('已清空所有账号。');
    }

    function clickLoginButton() {
        const loginBtn =
            document.querySelector('input[name="action_user_login"]') ||
            [...document.querySelectorAll('button, input[type="submit"], a')].find(el =>
                String(el.textContent || el.value || '').includes('ログインする')
            );

        if (loginBtn) {
            loginBtn.click();
            return true;
        }

        const form = document.querySelector('#login_area, form');
        if (form) {
            if (typeof form.requestSubmit === 'function') {
                form.requestSubmit();
            } else {
                form.submit();
            }
            return true;
        }

        return false;
    }

    function renderAccountSwitcher() {
        const old = document.getElementById('xserver-account-switcher');
        if (old) old.remove();

        const box = document.createElement('div');
        box.id = 'xserver-account-switcher';

        const title = document.createElement('div');
        title.className = 'xas-title';
        title.textContent = 'XServer 多账号';
        box.appendChild(title);

        const accounts = getAccounts();
        const currentIndex = getCurrentAccountIndex();

        const select = document.createElement('select');

        if (!accounts.length) {
            const option = document.createElement('option');
            option.value = '-1';
            option.textContent = '暂无账号：先输入后保存';
            select.appendChild(option);
        } else {
            accounts.forEach((acc, i) => {
                const option = document.createElement('option');
                option.value = String(i);
                option.textContent = `${i + 1}. ${acc.memberid}`;
                if (i === currentIndex) option.selected = true;
                select.appendChild(option);
            });
        }

        select.onchange = () => {
            const index = Number(select.value);
            if (!Number.isInteger(index) || index < 0) return;

            setCurrentAccountIndex(index);
            GM_setValue(SKIP_ONCE_KEY, true);

            const ok = fillLoginFormWithCurrentAccount();

            if (ok) {
                showToast('已切换账号，并填入登录框。');
            } else {
                showToast('已切换账号。若登录框未变化，请刷新页面。');
            }
        };

        box.appendChild(select);

        const autoLabel = document.createElement('label');
        const autoCheckbox = document.createElement('input');
        autoCheckbox.type = 'checkbox';
        autoCheckbox.checked = isAutoLoginEnabled();

        autoCheckbox.onchange = () => {
            setAutoLoginEnabled(autoCheckbox.checked);
            if (autoCheckbox.checked) {
                GM_setValue(SKIP_ONCE_KEY, false);
                showToast('已启用当前账号自动登录。');
            } else {
                showToast('自动登录已暂停，请手动登录。');
            }
        };

        autoLabel.appendChild(autoCheckbox);
        autoLabel.appendChild(document.createTextNode('启用当前账号自动登录'));
        box.appendChild(autoLabel);

        const saveBtn = document.createElement('button');
        saveBtn.className = 'btn-save';
        saveBtn.textContent = '保存当前输入为账号';
        saveBtn.onclick = () => {
            saveCurrentInputAsAccount();
        };
        box.appendChild(saveBtn);

        const fillBtn = document.createElement('button');
        fillBtn.className = 'btn-fill';
        fillBtn.textContent = '填入选中账号';
        fillBtn.onclick = () => {
            GM_setValue(SKIP_ONCE_KEY, true);
            const ok = fillLoginFormWithCurrentAccount();
            showToast(ok ? '已填入选中账号。' : '未找到登录输入框。');
        };
        box.appendChild(fillBtn);

        const manualBtn = document.createElement('button');
        manualBtn.className = 'btn-manual';
        manualBtn.textContent = '本次手动登录';
        manualBtn.onclick = () => {
            GM_setValue(SKIP_ONCE_KEY, true);
            fillLoginFormWithCurrentAccount();
            showToast('已填入当前账号，请手动点击登录。');
        };
        box.appendChild(manualBtn);

        const deleteBtn = document.createElement('button');
        deleteBtn.className = 'btn-delete';
        deleteBtn.textContent = '删除当前';
        deleteBtn.onclick = deleteCurrentAccount;
        box.appendChild(deleteBtn);

        const clearBtn = document.createElement('button');
        clearBtn.className = 'btn-clear';
        clearBtn.textContent = '清空所有账号';
        clearBtn.onclick = clearAllAccounts;
        box.appendChild(clearBtn);

        const note = document.createElement('div');
        note.className = 'xas-note';
        note.textContent = '提示：切换账号后会立刻填入登录框；避免马上自动登录，刷新一次即可用当前账号。';
        box.appendChild(note);

        document.body.appendChild(box);
    }

    async function handleLogin() {
        console.log(`${LOG_PREFIX} 当前在登录页面。`);
        updateStatusElement('正在处理登录...');

        renderAccountSwitcher();

        const current = getCurrentAccount();
        const hasError = !!document.querySelector('.errorMessage, .error, .alert');

        if (current) {
            fillLoginFormWithCurrentAccount();
        }

        const skipOnce = GM_getValue(SKIP_ONCE_KEY, false) === true;
        if (skipOnce) {
            GM_setValue(SKIP_ONCE_KEY, false);
            updateStatusElement('自动登录已暂停，请手动登录。');
            return;
        }

        if (current && isAutoLoginEnabled() && !hasError) {
            console.log(`${LOG_PREFIX} 发现当前账号，准备自动登录：${current.memberid}`);
            updateStatusElement('已检测到保存凭据，正在自动登录...');

            setTimeout(() => {
                try {
                    fillLoginFormWithCurrentAccount();
                    const ok = clickLoginButton();
                    if (!ok) throw new Error('找不到登录按钮或表单');
                } catch (e) {
                    console.error(`${LOG_PREFIX} 自动登录失败:`, e);
                    updateStatusElement('自动登录失败，请手动登录。');
                }
            }, 900);
        } else {
            console.log(`${LOG_PREFIX} 未启用自动登录或没有账号，等待手动操作。`);
            updateStatusElement('自动登录已暂停，请手动登录。');
        }

        const form = document.querySelector('#login_area, form');
        if (form) {
            form.addEventListener('submit', () => {
                try {
                    const { memberInput, passInput } = getLoginInputs();
                    const memberid = getInputValue(memberInput);
                    const user_password = passInput ? String(passInput.value || '') : '';

                    if (memberid && user_password) {
                        addOrUpdateAccount(memberid, user_password);
                        console.log(`${LOG_PREFIX} 已保存/更新账号：${memberid}`);
                    }
                } catch (e) {
                    console.error(`${LOG_PREFIX} 保存凭据时出错:`, e);
                }
            });
        }
    }

    function handleVPSDashboard() {

        updateStatusElement('正在检查续期状态...');

        try {
            const tomorrow = new Date(Date.now() + 86400000).toLocaleDateString('sv', { timeZone: 'Asia/Tokyo' });

                updateStatusElement('未找到免费VPS。');
                return;
            }

            const expireSpan = row.querySelector('.contract__term');
            const expireDate = expireSpan ? expireSpan.textContent.trim() : null;

            console.log(`${LOG_PREFIX} 页面上的到期日: ${expireDate || '未找到'}`);
            console.log(`${LOG_PREFIX} 明天的日期: ${tomorrow}`);

            if (expireDate === tomorrow) {
                console.log(`${LOG_PREFIX} 条件满足：到期日为明天。正在跳转到续期页面...`);
                const detailLink = row.querySelector('a[href^="/xapanel/xvps/server/detail?id="]');

                    updateStatusElement('检测到即将过期，正在续期...');
                    setTimeout(() => {
                        location.href = detailLink.href.replace('detail?id', 'freevps/extend/index?id_vps');

                    throw new Error('无法定位续期链接');

                updateStatusElement('当前VPS无需续期。');
                setTimeout(removeStatusElement, 3000);
            }
        } catch (e) {
            console.error(`${LOG_PREFIX} 在VPS管理主页处理出现错误:`, e);
            updateStatusElement('检查续期状态出错，请刷新页面重试。');
        }
    }

    function decodeCaptcha(predictionTensor) {
        const blankIndex = predictionTensor.shape[2] - 1;
        const bestPath = predictionTensor.argMax(-1).dataSync();
        const digits = [];
        let previous = blankIndex;

        for (const index of bestPath) {
            if (index !== blankIndex && index !== previous) {
                digits.push(String(index));
            }
            previous = index;
        }

        return digits.join('');
    }

    async function imageToCode(img) {
        await tf.ready();

        const model = await tf.loadLayersModel(
            'https://github30.github.io/captcha-cloudrun/web_model/model.json?v=20260407-2'
        );

        return tf.tidy(() => {
            const tensor = tf.browser
                .fromPixels(img)
                .resizeBilinear([60, 300])
                .toFloat()
                .div(255)
                .expandDims(0);

            return decodeCaptcha(model.predict(tensor));
        });
    }

    async function handleCaptchaPage() {

        updateStatusElement('正在识别并输入验证码...');

        try {
            await waitForDOMReady();

            const img =
                document.querySelector('img[src^="data:image"]') ||
                document.querySelector('img[src^="data:"]');

            if (!img || !img.src) {
                throw new Error('未找到验证码图片');
            }

            console.log(`${LOG_PREFIX} 已找到验证码图片，正在识别...`);
            updateStatusElement('正在识别验证码，请稍候...');

            let codeResponse = '';
            const maxRetries = 3;

            for (let retryCount = 0; retryCount < maxRetries; retryCount++) {
                try {
                    codeResponse = await imageToCode(img);
                    if (codeResponse && codeResponse.length >= 4) break;
                    throw new Error('验证码识别结果无效');
                } catch (err) {
                    if (retryCount >= maxRetries - 1) throw err;
                    console.log(`${LOG_PREFIX} 验证码识别失败，重试 ${retryCount + 1}/${maxRetries}`);
                }
            }

            const code = String(codeResponse || '').trim();

            if (!code || code.length < 4) {
                throw new Error('未接收到有效验证码或验证码太短');
            }

            console.log(`${LOG_PREFIX} 验证码: ${code}`);
            updateStatusElement('验证码识别完成，准备提交表单...');

            const input =
                document.querySelector('[placeholder*="上の画像"]') ||
                document.querySelector('input[name*="captcha"], input[type="text"]');

            if (!input) {
                throw new Error('未找到验证码输入框');
            }

            setNativeValue(input, code);

            console.log(`${LOG_PREFIX} 已将验证码填入输入框。`);
            updateStatusElement('已完成验证码填写，正在处理人机验证...');

            const cf =
                document.querySelector('[name="cf-turnstile-response"]') ||
                document.querySelector('textarea[name="cf-turnstile-response"]');

            if (!cf) {
                console.warn(`${LOG_PREFIX} 未检测到 Cloudflare Turnstile，直接提交。`);
                submitForm();
                return;
            }

            if (cf.value) {
                console.log(`${LOG_PREFIX} Cloudflare 令牌已存在，直接提交。`);
                submitForm();
                return;
            }

            console.log(`${LOG_PREFIX} 等待用户手动完成 Cloudflare 验证...`);
            updateStatusElement('请手动完成 Cloudflare 验证...');

            const observer = new MutationObserver(() => {
                if (cf.value) {
                    console.log(`${LOG_PREFIX} Cloudflare 令牌已生成，准备提交。`);
                    observer.disconnect();
                    submitForm();
                }
            });

            observer.observe(cf, {
                attributes: true,
                attributeFilter: ['value'],
            });

            const interval = setInterval(() => {
                if (cf.value) {
                    clearInterval(interval);
                    observer.disconnect();
                    console.log(`${LOG_PREFIX} Cloudflare 令牌已生成，准备提交。`);
                    submitForm();
                }
            }, 1000);

        } catch (error) {
            console.error(`${LOG_PREFIX} 处理验证码时发生错误:`, error);
            updateStatusElement('验证码处理异常，请刷新页面重试。');
        }

        function submitForm() {
            updateStatusElement('所有验证已完成，准备提交...');

            setTimeout(() => {
                const submitBtn =
                    [...document.querySelectorAll('button, input[type="submit"], a')].find(el =>
                        String(el.textContent || el.value || '').includes('無料VPSの利用を継続する')
                    ) ||
                    document.querySelector('input[type="submit"], button[type="submit"]');

                if (submitBtn) {
                    submitBtn.click();
                } else {
                    console.error(`${LOG_PREFIX} 未找到可点击的提交按钮`);
                    updateStatusElement('找不到提交按钮，请手动提交表单');
                }
            }, 1000);
        }
    }

    function main() {
        if (isRunning) return;
        isRunning = true;

        const path = window.location.pathname;

        if (path.startsWith('/xapanel/login/xvps')) {
            handleLogin();
        } else if (path.includes('/xapanel/xvps/index')) {
            handleVPSDashboard();
        } else if (path.includes('/xapanel/xvps/server/freevps/extend/index')) {
            handleRenewalPage();
        } else if (
            path.includes('/xapanel/xvps/server/freevps/extend/conf') ||
            path.includes('/xapanel/xvps/server/freevps/extend/do')
        ) {
            handleCaptchaPage();
        } else {
            console.log(`${LOG_PREFIX} 当前不在已支持的路径中，脚本不会执行任何操作。`);
            isRunning = false;
        }
    }

    main();
})();
