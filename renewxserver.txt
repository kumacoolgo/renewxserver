db.js
import Database from 'better-sqlite3'


const db = new Database('/data/renewxserver.sqlite')


db.exec(`
CREATE TABLE IF NOT EXISTS accounts (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  name TEXT NOT NULL,
  email TEXT NOT NULL,
  password TEXT NOT NULL,
  expire_date TEXT,
  last_checked_at TEXT,
  created_at TEXT DEFAULT CURRENT_TIMESTAMP
);
`)


export function addAccount(name, email, password) {
  const stmt = db.prepare('INSERT INTO accounts (name, email, password) VALUES (?, ?, ?)')
  return stmt.run(name, email, password)
}


export function listAccounts() {
  return db.prepare('SELECT * FROM accounts ORDER BY id DESC').all()
}


export function getAccount(id) {
  return db.prepare('SELECT * FROM accounts WHERE id = ?').get(id)
}


export function deleteAccount(id) {
  return db.prepare('DELETE FROM accounts WHERE id = ?').run(id)
}


export function updatePassword(id, password) {
  return db.prepare('UPDATE accounts SET password = ? WHERE id = ?').run(password, id)
}


export function updateExpireDate(id, expireDate) {
  return db.prepare('UPDATE accounts SET expire_date = ?, last_checked_at = CURRENT_TIMESTAMP WHERE id = ?').run(expireDate, id)
}
xserver.js
import { chromium } from 'playwright'


const LOGIN_URL = 'https://secure.xserver.ne.jp/xapanel/login/xvps/'
const INDEX_URL = 'https://secure.xserver.ne.jp/xapanel/xvps/index'


export function daysLeft(expireDate) {
  const today = new Date().toLocaleDateString('sv', { timeZone: 'Asia/Tokyo' })
  const a = new Date(today + 'T00:00:00+09:00')
  const b = new Date(expireDate + 'T00:00:00+09:00')
  return Math.ceil((b - a) / 86400000)
}


export async function checkExpire(account) {
  const browser = await chromium.launch({ headless: true })
  const page = await browser.newPage({ viewport: { width: 1080, height: 1024 } })


  try {
    await page.goto(LOGIN_URL, { waitUntil: 'networkidle' })
    await page.locator('#memberid').fill(account.email)
    await page.locator('#user_password').fill(account.password)
    await page.locator('text=ログインする').click()
    await page.waitForLoadState('networkidle')


    await page.goto(INDEX_URL, { waitUntil: 'networkidle' })


    const expireDate = await page
      .locator('tr:has(.freeServerIco) .contract__term')
      .first()
      .textContent({ timeout: 15000 })


    const clean = expireDate.trim()
    return {
      expireDate: clean,
      daysLeft: daysLeft(clean),
    }
  } finally {
    await browser.close()
  }
}

async function checkOne(chatId, id) {
  const acc = getAccount(id)
  if (!acc) return bot.sendMessage(chatId, '账号不存在')


  const m = await bot.sendMessage(chatId, `正在检测 ${acc.name} ...`)
  try {
    const result = await checkExpire(acc)
    updateExpireDate(id, result.expireDate)
    const warn = result.daysLeft <= 1 ? '⚠️ 小于等于 1 天，必须更新' : '✅ 正常'
    return bot.editMessageText(
      `${acc.name}
期限：${result.expireDate}
剩余：${result.daysLeft} 天
状态：${warn}`,
      { chat_id: chatId, message_id: m.message_id }
    )
  } catch (e) {
    return bot.editMessageText(`检测失败：${e.message}`, {
      chat_id: chatId,
      message_id: m.message_id,
    })
  }
}


async function checkAll(chatId) {
  const accounts = listAccounts()
  if (!accounts.length) return bot.sendMessage(chatId, '暂无账号')


  for (const acc of accounts) {
    await checkOne(chatId, acc.id)
  }
}
